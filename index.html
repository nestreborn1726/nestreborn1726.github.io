abstract class WeaponSkill extends Skill {
    abstract public boolean useSkill(Player player, Enemy enemy);
    
    static class Slash extends WeaponSkill {
        @Override
        public boolean useSkill(Player player, Enemy enemy) {
            if(!isOnCooldown()) {
                // Modular Debuff Application
                Debuff bleedDebuff = new Debuff.Bleed();
                bleedDebuff.setEffect(true);
                bleedDebuff.setNumberOfTurn(3);
                enemy.addDebuff(bleedDebuff);

                int damage = (int) (player.getCurrentAttack() * 1.3);
                int newHp = enemy.getHp() - damage;
                enemy.setHp(newHp);
                
                // State Management
                setCooldown(3);
                setOnCooldown(true);
                return true;
            }
            return false;
        }
    }
}
public static double calculateBlockedDamage(int attack, int defense) {
    double damagedBlocked;
    if (defense >= attack) {
        double value = (double) defense / attack;
        value = 100.0 / value;
        double percentage = value / 100.0;
        damagedBlocked = percentage;
    } else {
        double value = defense / attack;
        damagedBlocked = value;
    }
    return damagedBlocked;
}
